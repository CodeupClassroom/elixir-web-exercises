PROBLEM:
	"bob sally lou frank louise tom"
capitalize all of the words and output the resulting string
	"Bob Sally Lou Frank Louise Tom"

var s1 = "bob sally lou frank louise tom";
var a3 = s1.split(" ");
for (let i = 0; i < a3.length; i++) {
    a3[i] = a3[i].charAt(0).toUpperCase() + a3[i].substring(1);
}
console.log(a3.join(" "));
